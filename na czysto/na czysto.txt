def executeFunction(fileName,generalCheck,mechanicalCheck,thermalCheck):
	odb = openOdb(fileName)
	global odb
	listMaterials = odb.materials.keys()
	currentDate = datetime.datetime.now()
	
	myGetName.getName(odb.name)
	print "Date: " + currentDate.strftime("%Y-%m-%d %H:%M")
	for tmp in listMaterials:
		basicInfoFun(tmp)
		if generalCheck == 1:
			generalInfoFun(tmp)
		if mechanicalCheck == 1:
			mechanicalInfoFun(tmp)
		if thermalCheck == 1:
			thermalInfoFun(tmp)
		print " "
		print " "
		print " "
